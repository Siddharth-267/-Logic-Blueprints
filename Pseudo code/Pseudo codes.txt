16. Fibonacci series
SET n = 10
SET prev = 0
SET next = 1
SET sum = 0

PRINT prev, next

WHILE sum <= n
SET sum = prev + next
IF sum <= n THEN
PRINT sum
SET prev = next
SET next = sum
END IF
END WHILE
END
